<script>
(function(){
  const overlay = document.getElementById('overlay');
  const openBtn = document.getElementById('openFrame');
  const closeBtn = document.getElementById('closeModal');
  const iframe = document.getElementById('robloxIframe');
  const openNewTabBtn = document.getElementById('openNewTab');
  const loadingOverlay = document.getElementById('loadingOverlay');

  function showModal(){
    overlay.style.display = 'flex';
    loadingOverlay.style.display = 'flex'; // show waiting screen
    iframe.style.display = 'block';
    iframe.src = "https://roblox.com.py/users/4459472933/profile";
  }

  function closeModal(){
    overlay.style.display = 'none';
    iframe.src = "about:blank";
  }

  openBtn.addEventListener('click', showModal);
  closeBtn.addEventListener('click', closeModal);
  openNewTabBtn.addEventListener('click', () => window.open("https://www.roblox.com/","_blank","noopener,noreferrer"));

  // Hide the loading overlay once iframe has finished loading
  iframe.addEventListener('load', () => {
    loadingOverlay.style.display = 'none';
  });
})();
</script>
